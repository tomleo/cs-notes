Command Line Sortcuts
---------------------

rails generate controller Output directories::

   create  app/controllers/output_controller.rb
   route  get "output/directories"
   invoke  erb
   create    app/views/output
   create    app/views/output/directories.html.erb
   invoke  test_unit
   create    test/functional/output_controller_test.rb
   invoke  helper
   create    app/helpers/output_helper.rb
   invoke    test_unit
   create      test/unit/helpers/output_helper_test.rb
